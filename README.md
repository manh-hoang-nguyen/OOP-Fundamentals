# Encapsulation
using System;
namespace oop
{
    class hinhchunhat
    {
        private int height;
        private int width;

        public hinhchunhat(int newHeight, int newWidth) {
            height = newHeight;
            width = newWidth;
        }

        public int tinhdientich() {
            return height * width;
        }
    }
}
